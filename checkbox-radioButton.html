Checkboxes

Selenium Test Code - Checkboxes

@Test
public void checkBoxTest() {

    // Finding checkbox elements
    WebElement sportsCheckBox = driver.findElement(By.id("sports"));
    WebElement musicCheckBox = driver.findElement(By.id("music"));
    WebElement readingCheckBox = driver.findElement(By.id("reading"));
    WebElement travelCheckBox = driver.findElement(By.id("travel"));
    WebElement cookingCheckBox = driver.findElement(By.id("cooking"));
    WebElement getCheckBoxButton = driver.findElement(By.xpath("//button[@onclick='getCheckboxValues()']"));
    WebElement resultBox = driver.findElement(By.id("checkbox-result"));

    // Verify elements are displayed
    Assertions.assertTrue(sportsCheckBox.isDisplayed());
    Assertions.assertTrue(musicCheckBox.isDisplayed());
    Assertions.assertTrue(readingCheckBox.isDisplayed());
    Assertions.assertTrue(travelCheckBox.isDisplayed());
    Assertions.assertTrue(cookingCheckBox.isDisplayed());

    // Select checkboxes if not already selected
    if (!musicCheckBox.isSelected()){
        musicCheckBox.click();
    }

    if (!cookingCheckBox.isSelected()){
        cookingCheckBox.click();
    }

    if (!travelCheckBox.isSelected()){
        travelCheckBox.click();
    }

    getCheckBoxButton.click();

    // Verify result
    String expectedResult = "Selected interests: music, travel, cooking";
    String actualResult = resultBox.getText();

    Assertions.assertEquals(expectedResult, actualResult);
}

Text Input Fields

Selenium Test Code - Text Input Fields

@Test
public void textInputFieldsTest() throws InterruptedException {

    // Finding text input elements
    WebElement usernameTextBox = driver.findElement(By.id("username"));
    WebElement emailTextBox = driver.findElement(By.id("email"));
    WebElement passwordTextBox = driver.findElement(By.id("password"));
    WebElement ageTextBox = driver.findElement(By.id("number"));
    WebElement getTextValuesButton = driver.findElement(By.xpath("//button[@onclick='getTextValues()']"));
    WebElement clearFieldsButton = driver.findElement(By.xpath("//button[@onclick='clearTextFields()']"));
    WebElement textResult = driver.findElement(By.id("text-result"));

    // Enter text values
    usernameTextBox.sendKeys("john");
    emailTextBox.sendKeys("john@gmaill.com");
    passwordTextBox.sendKeys("john123?");
    ageTextBox.sendKeys("33");

    getTextValuesButton.click();

    // Verify entered values
    String expectedResult = "Username: john\n" +
             "Email: john@gmailll.com\n" +
             "Password: ***\n" +
             "Age: 33";

    String actualResult = textResult.getText();

    Assertions.assertEquals(expectedResult, actualResult);

    Thread.sleep(3000);

    // Test clear button
    clearFieldsButton.click();

    String clearedExpectedResult = "Username: \n" +
            "Email: \n" +
            "Password: \n" +
            "Age: ";

    String clearedActualResult = textResult.getText();

    Assertions.assertEquals(clearedExpectedResult, clearedActualResult);
}

Selenium Test Code - Toggle Switch

@Test
public void toggleSwitchTest() {

    // Finding toggle switch elements
    WebElement toggleSwitch = driver.findElement(By.id("toggle"));
    WebElement toggleStatus = driver.findElement(By.id("toggle-status"));

    // Verify initial state
    Assertions.assertFalse(toggleSwitch.isSelected());
    Assertions.assertEquals("Disabled", toggleStatus.getText());

    // Toggle switch on
    toggleSwitch.click();

    // Verify enabled state
    Assertions.assertTrue(toggleSwitch.isSelected());
    Assertions.assertEquals("Enabled", toggleStatus.getText());

    // Toggle switch off
    toggleSwitch.click();

    // Verify disabled state
    Assertions.assertFalse(toggleSwitch.isSelected());
    Assertions.assertEquals("Disabled", toggleStatus.getText());
}

Range Slider

Selenium Test Code - Range Slider

@Test
public void rangeSliderTest() throws InterruptedException {

    // Finding range slider elements
    WebElement volumeSlider = driver.findElement(By.id("range"));
    WebElement brightnessSlider = driver.findElement(By.id("brightness"));
    WebElement volumeValue = driver.findElement(By.id("range-value"));
    WebElement brightnessValue = driver.findElement(By.id("brightness-value"));

    // Verify initial values
    Assertions.assertEquals("50", volumeSlider.getAttribute("value"));
    Assertions.assertEquals("75", brightnessSlider.getAttribute("value"));

    // Using Actions class to move slider
    Actions actions = new Actions(driver);

    // Move volume slider to 80
    actions.clickAndHold(volumeSlider)
           .moveByOffset(60, 0)
           .release()
           .perform();

    Thread.sleep(1000);

    // Verify volume slider changed
    String volumeAfter = volumeSlider.getAttribute("value");
    Assertions.assertNotEquals("50", volumeAfter);

    // Alternative: Using sendKeys for precise value
    volumeSlider.clear();
    volumeSlider.sendKeys("90");
    
    Assertions.assertEquals("90", volumeSlider.getAttribute("value"));
}

Form Actions

Selenium Test Code - Form Actions

@Test
public void formSubmissionTest() {

    // Fill out form with sample data
    driver.findElement(By.id("sports")).click();
    driver.findElement(By.id("music")).click();
    driver.findElement(By.id("email-contact")).click();
    driver.findElement(By.id("advanced")).click();
    
    WebElement usernameField = driver.findElement(By.id("username"));
    usernameField.clear();
    usernameField.sendKeys("testuser");
    
    WebElement emailField = driver.findElement(By.id("email"));
    emailField.clear();
    emailField.sendKeys("test@example.com");
    
    WebElement commentsField = driver.findElement(By.id("comments"));
    commentsField.clear();
    commentsField.sendKeys("Test comment");
    
    // Submit form
    WebElement submitButton = driver.findElement(By.xpath("//button[@onclick='submitForm()']"));
    WebElement formResult = driver.findElement(By.id("form-result"));
    
    submitButton.click();
    
    // Verify form submission
    String result = formResult.getText();
    Assertions.assertTrue(result.contains("Form submitted with data:"));
    Assertions.assertTrue(result.contains("testuser"));
    Assertions.assertTrue(result.contains("test@example.com"));
}

Test Execution Guidelines

Common Setup Requirements:

  • Browser: Chrome WebDriver
  • Timeouts: Implicit wait 15 seconds, Explicit wait 10 seconds
  • Window State: Maximized
  • Page Load: Complete before test execution

Best Practices:

  • Use WebDriverWait instead of Thread.sleep()
  • Clear fields before entering data
  • Verify element visibility before interaction
  • Use explicit waits for dynamic content
  • Proper cleanup after each test
